function getCurrentDay() { const daysOfWeek = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"]; const currentDate = new Date(); const dayIndex = currentDate.getDay(); return daysOfWeek[dayIndex]; } function getCurrentTime() { const currentDate = new Date(); const hours = currentDate.getHours(); const minutes = currentDate.getMinutes(); const seconds = currentDate.getSeconds(); const period = hours >= 12 ? "PM" : "AM"; const formattedHours = hours % 12 || 12; // Convert to 12-hour format return `${formattedHours}:${minutes}:${seconds} ${period}`; } const day = getCurrentDay(); const time = getCurrentTime(); console.log(`Today is: ${day}.`); console.log(`Current time is: ${time}`);
